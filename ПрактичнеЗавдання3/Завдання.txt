#include <stdio.h>

// Функція для знаходження кількості чисел з розрядами, де три однакові цифри не стоять поруч
long long countNumbers(int r) {
    long long dp[r + 1][2];
    dp[1][0] = 2;  // Кількість чисел з 1 розрядом, починаючи з 5
    dp[1][1] = 2;  // Кількість чисел з 1 розрядом, починаючи з 9

    for (int i = 2; i <= r; i++) {
        dp[i][0] = dp[i - 1][0] + dp[i - 1][1];  // Кількість чисел, починаючи з 5
        dp[i][1] = dp[i - 1][0] + dp[i - 1][1];  // Кількість чисел, починаючи з 9
    }

    return dp[r][0] + dp[r][1];
}

int main() {
    int r;

    // Зчитування цілого числа р
    printf("Введіть ціле число р: ");
    scanf("%d", &r);

    // Знаходження кількості чисел з розрядами
    long long count = countNumbers(r);

    // Виведення результату
    printf("Кількість чисел з %d розрядами: %lld\n", r, count);

    return 0;
}
